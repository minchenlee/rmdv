\section{The Lorentz Transformation}
For a frame moving with velocity $v$ along $x$, coordinates transform as:

\begin{equation}
x' = \frac{x - vt}{\sqrt{1 - v^2/c^2}}
\end{equation}

\begin{equation}
t' = \frac{t - vx/c^2}{\sqrt{1 - v^2/c^2}}
\end{equation}

\section{Mass--Energy Equivalence}
The rest energy of a body is

\begin{equation}
E = mc^2
\end{equation}

and including momentum $p$,

\begin{equation}
E^2 = (pc)^2 + (mc^2)^2.
\end{equation}
